import urllib

# http://docs.python.jp/3/library/urllib.parse.html
# python2とは違う
quoted = urllib.parse.quote("あいうえお")
print(quoted)
# %E3%81%82%E3%81%84%E3%81%86%E3%81%88%E3%81%8A
print(urllib.parse.unquote(quoted))
# あいうえお

# URLを各部分にparseしてくれて便利
