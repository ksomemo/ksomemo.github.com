@ipywidgets.interact(
    a=(-10, 10, 1),
    b={'a': 1, 'b': 2.0, 'c': 5.0},
    check=True,
)
def plot_linear_func(a=10, b=1, check=False):
    print("a =", a)
    print("b =", b)
    print("check =", check)

    x = pd.Series(range(10))
    y = a * x + check * b
    plt.plot(y)
