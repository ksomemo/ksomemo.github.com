import numpy as np
import pandas as pd
import matplotlib
import matplotlib.pyplot as plt
plt.style.use('ggplot')
%matplotlib inline


df_ex = pd.DataFrame(
    {
        "a": np.random.randint(0, 3+1, 50),
        "b": range(50),
        "c": list("abcde") * 10,
        "y": np.random.randint(0, 1+1, 50)
    }
)

patsy.dmatrices("y ~ .", df_ex, return_type="dataframe")
