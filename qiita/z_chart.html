import numpy as np
import pandas as pd


np.random.seed(0)
sales = pd.DataFrame({
    "sale" :np.random.randint(10, high=15, size=24),
}, index=pd.date_range("2015-01-01", "2016-12-01", freq="MS"))

sales["cumsum_12month"] = sales["sale"].rolling(window=12).sum()
sales["2016_cumsum"] = sales["2016-01-01":]["sale"].cumsum()
sales.plot()
