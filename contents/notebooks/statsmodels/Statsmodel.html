    'figure.figsize': (10, 8),
    'font.size': 18,

    'figure.dpi': 100,
    'axes.labelsize': 18,
    'axes.titlesize': 20,
    'xtick.labelsize': 15,
    'ytick.labelsize': 20,
    'legend.fontsize': 15,
    'lines.markersize': 15,
}
with plt.rc_context(plot_context):
    fig, rects = mosaic(iris_copy, ["sepal_length_round", "species"], gap=0.01)
