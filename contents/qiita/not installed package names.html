not installed package names¶

installed.pkg.names <- rownames(installed.packages())
pkg.names <- (read.table(pipe("pbpaste"), header = F, stringsAsFactors = F))$V1
not.installed.pkg.names <- setdiff(pkg.names, installed.pkg.names)
