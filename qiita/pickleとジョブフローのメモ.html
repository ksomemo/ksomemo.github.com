Pickle化するコードと読み込み¶

tree
.
├── item.py
├── pickle_sample.py
├── user.py
└── users.pickle

import pickle
import user
import item

user1 = user.User(1, 'A')
user2 = user.User(2, 'B')
item1 = item.Item('I1')
item2 = item.Item('I2')
user1.append_items([item1, item2])
user2.append_items([item1, item2])
users = [user1, user2]
print(users)

with open('users.pickle', 'wb') as f:
    pickle.dump(users, f)
with open('users.pickle', 'rb') as f:
    users_loaded = pickle.load(f)
    print(users_loaded)

class User:
    def __init__(self, id, name):
        self.id = id
        self.name = name
        self.items = []

    def __repr__(self):
        return '<%s(%s): %s>' % (self.__class__, id(self), self.__dict__)

    def append_items(self, items):
        for i in items:
            self.items.append(i)

class Item:
    def __init__(self, name):
        self.name = name

    def __repr__(self):
        return '<%s(%s): %s>' % (self.__class__, id(self), self.__dict__)
