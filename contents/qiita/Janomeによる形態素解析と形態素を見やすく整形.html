# 見づらい
from janome.tokenizer import Tokenizer
t = Tokenizer()
for token in t.tokenize(u'すもももももももものうち'):
    print(token)
    print(type(token))
    for attr_name in dir(token):
        if attr_name.startswith("_"):
            continue
        attr = getattr(token, attr_name)
        print(attr_name, attr, type(attr))
    break
