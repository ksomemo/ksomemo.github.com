import pandas as pd
import numpy as np
import io

def from_string(_str, **kwargs):
    # http://pandas.pydata.org/pandas-docs/stable/generated/pandas.DataFrame.from_csv.html
    return pd.DataFrame.from_csv(io.StringIO(_str), **kwargs)

_csv = """a,b,c
1,2,3
4,5,6
"""
